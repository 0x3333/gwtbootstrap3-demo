023import com.google.gwt.dom.client.Element;
024import com.google.gwt.safehtml.shared.SafeHtml;
025import com.google.gwt.user.client.Event;
026import org.gwtbootstrap3.client.shared.event.AlertCloseEvent;
027import org.gwtbootstrap3.client.shared.event.AlertClosedEvent;
028import org.gwtbootstrap3.client.ui.base.button.CloseButton;
029import org.gwtbootstrap3.client.ui.base.helper.StyleHelper;
030import org.gwtbootstrap3.client.ui.constants.AlertType;
031import org.gwtbootstrap3.client.ui.constants.ButtonDismiss;
032import org.gwtbootstrap3.client.ui.constants.Styles;
033
034/**
035 * Alert block.
036 * <p/>
037 * Use {@link #setDismissable(boolean)} to add a close ("x") button.
038 *
039 * @author Sven Jacobs
040 * @author Joshua Godi
041 * @see org.gwtbootstrap3.client.shared.event.AlertCloseEvent
042 * @see org.gwtbootstrap3.client.shared.event.AlertClosedEvent
043 */
044public class Alert extends HTMLPanel implements HasType<AlertType>, HasResponsiveness {
045    private static final String CLOSE = "close";
046
047    private final CloseButton closeButton = new CloseButton();
048
049    public Alert(final String html) {
050        super(html);
051
052        setStyleName(Styles.ALERT);
053        setType(AlertType.WARNING);
054
055        closeButton.setDismiss(ButtonDismiss.ALERT);
056    }
057
058    public Alert(final String html, final AlertType type) {
059        this(html);
060        setType(type);
061    }
062
063    public Alert(final SafeHtml safeHtml) {
064        this(safeHtml.asString());
065    }
066
067    public Alert(final SafeHtml safeHtml, final AlertType type) {
068        this(safeHtml.asString(), type);
069    }
070
071    @Override
072    protected void onLoad() {
073        super.onLoad();
074        bindJavaScriptEvents(getElement());
075    }
076
077    /**
078     * Sets alert type.
079     *
080     * @param type Alert type
081     * @see AlertType
082     */
083    @Override
084    public void setType(final AlertType type) {
085        StyleHelper.addUniqueEnumStyleName(this, AlertType.class, type);
086    }
087
088    @Override
089    public AlertType getType() {
090        return AlertType.fromStyleName(getStyleName());
091    }
092
093    /**
094     * Adds a close button to the alert
095     *
096     * @param dismissable Adds close button when {@code true}
097     */
098    public void setDismissable(final boolean dismissable) {
099        if (dismissable) {
100            insert(closeButton, getElement(), 0, true);
101            addStyleName(Styles.ALERT_DISMISSABLE);
102        } else {
103            closeButton.removeFromParent();
104            removeStyleName(Styles.ALERT_DISMISSABLE);
105        }
106    }
107
108    public boolean isDismissable() {
109        return closeButton.getParent() != null;
110    }
111
112    /**
113     * Closes alert.
114     */
115    public void close() {
116        alert(getElement(), CLOSE);
117    }
118
119    protected void onClose(final Event evt) {
120        fireEvent(new AlertCloseEvent(evt));
121    }
122
123    protected void onClosed(final Event evt) {
124        fireEvent(new AlertClosedEvent(evt));
125    }
126
127    @Override
128    public void setVisibleOn(final String deviceSizeString) {
129        StyleHelper.setVisibleOn(this, deviceSizeString);
130    }
131
132    @Override
133    public void setHiddenOn(final String deviceSizeString) {
134        StyleHelper.setHiddenOn(this, deviceSizeString);
135    }
136
137    // @formatter:off
138
139    private native void alert(final Element e, final String arg) /*-{
140        $wnd.jQuery(e).alert(arg);
141    }-*/;
142
143    private native void bindJavaScriptEvents(final Element e) /*-{
144        var target = this;
145        var $alert = $wnd.jQuery(e);
146
147        $alert.on('close.bs.alert', function (evt) {
148            target.@org.gwtbootstrap3.client.ui.Alert::onClose(Lcom/google/gwt/user/client/Event;)(evt);
149        });
150
151        $alert.on('closed.bs.alert', function (evt) {
152            target.@org.gwtbootstrap3.client.ui.Alert::onClosed(Lcom/google/gwt/user/client/Event;)(evt);
153        });
154    }-*/;
155}
