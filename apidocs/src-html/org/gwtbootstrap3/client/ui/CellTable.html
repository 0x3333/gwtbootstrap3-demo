023import com.google.gwt.core.client.GWT;
024import com.google.gwt.resources.client.ImageResource;
025import com.google.gwt.user.client.ui.Widget;
026import com.google.gwt.view.client.ProvidesKey;
027import org.gwtbootstrap3.client.ui.base.helper.StyleHelper;
028import org.gwtbootstrap3.client.ui.constants.TableType;
029
030/**
031 * @author Joshua Godi
032 */
033public class CellTable<T> extends com.google.gwt.user.cellview.client.CellTable<T> implements HasResponsiveness {
034    private static final int DEFAULT_PAGESIZE = 15;
035    private static Resources DEFAULT_RESOURCES;
036
037    /**
038     * Constructs a table with a default page size of 15.
039     */
040    public CellTable() {
041        this(DEFAULT_PAGESIZE);
042    }
043
044    /**
045     * Constructs a table with the given page size.
046     *
047     * @param pageSize the page size
048     */
049    public CellTable(final int pageSize) {
050        this(pageSize, getDefaultResources());
051    }
052
053    /**
054     * Constructs a table with a default page size of 15, and the given
055     * {@link ProvidesKey key provider}.
056     *
057     * @param keyProvider an instance of ProvidesKey<T>, or null if the record
058     *                    object should act as its own key
059     */
060    public CellTable(final ProvidesKey<T> keyProvider) {
061        this(DEFAULT_PAGESIZE, keyProvider);
062    }
063
064    /**
065     * Constructs a table with the given page size with the specified
066     * {@link Resources}.
067     *
068     * @param pageSize  the page size
069     * @param resources the resources to use for this widget
070     */
071    public CellTable(final int pageSize, final Resources resources) {
072        this(pageSize, resources, null);
073    }
074
075    /**
076     * Constructs a table with the given page size and the given
077     * {@link ProvidesKey key provider}.
078     *
079     * @param pageSize    the page size
080     * @param keyProvider an instance of ProvidesKey<T>, or null if the record
081     *                    object should act as its own key
082     */
083    public CellTable(final int pageSize, final ProvidesKey<T> keyProvider) {
084        this(pageSize, getDefaultResources(), keyProvider);
085    }
086
087    /**
088     * Constructs a table with the given page size, the specified
089     * {@link Resources}, and the given key provider.
090     *
091     * @param pageSize    the page size
092     * @param resources   the resources to use for this widget
093     * @param keyProvider an instance of ProvidesKey<T>, or null if the record
094     *                    object should act as its own key
095     */
096    public CellTable(final int pageSize, final Resources resources, final ProvidesKey<T> keyProvider) {
097        this(pageSize, resources, keyProvider, createDefaultLoadingIndicator(resources));
098    }
099
100    /**
101     * Constructs a table with the specified page size, {@link Resources}, key
102     * provider, and loading indicator.
103     *
104     * @param pageSize         the page size
105     * @param resources        the resources to use for this widget
106     * @param keyProvider      an instance of ProvidesKey<T>, or null if the record
107     *                         object should act as its own key
108     * @param loadingIndicator the widget to use as a loading indicator, or null
109     *                         to disable
110     */
111    public CellTable(final int pageSize, final Resources resources, final ProvidesKey<T> keyProvider,
112                     final Widget loadingIndicator) {
113        this(pageSize, resources, keyProvider, loadingIndicator, true, true);
114    }
115
116    /**
117     * Constructs a table with the specified page size, {@link Resources}, key
118     * provider, and loading indicator.
119     *
120     * @param pageSize           the page size
121     * @param resources          the resources to use for this widget
122     * @param keyProvider        an instance of ProvidesKey<T>, or null if the record
123     *                           object should act as its own key
124     * @param loadingIndicator   the widget to use as a loading indicator, or null
125     *                           to disable
126     * @param enableColGroup     enable colgroup element. This is used when the table is using fixed
127     *                           layout and when column style is added. Ignoring this element will boost rendering
128     *                           performance. Note that when colgroup is disabled, {@link #setColumnWidth}
129     * @param attachLoadingPanel attaching the table section that contains the empty table widget and
130     *                           the loading indicator. Attaching this to the table significantly improve the rendering
131     *                           performance in webkit based browsers but also introduces significantly larger latency
132     *                           in IE. If the panel is not attached to the table, it won't be displayed. But the user
133     *                           can call {@link #getTableLoadingSection} and attach it to other elements outside the
134     *                           table element
135     */
136    public CellTable(final int pageSize, final Resources resources, final ProvidesKey<T> keyProvider,
137                     final Widget loadingIndicator, final boolean enableColGroup, final boolean attachLoadingPanel) {
138        super(pageSize, resources, keyProvider, loadingIndicator, enableColGroup, attachLoadingPanel);
139        StyleHelper.addEnumStyleName(this, TableType.DEFAULT);
140    }
141
142    /**
143     * Create the default loading indicator using the loading image in the
144     * specified {@link Resources}.
145     *
146     * @param resources the resources
147     * @return a widget loading indicator
148     */
149    private static Widget createDefaultLoadingIndicator(final Resources resources) {
150        final ImageResource loadingImg = resources.cellTableLoading();
151        return (loadingImg == null) ? null : new com.google.gwt.user.client.ui.Image(loadingImg);
152    }
153
154    private static Resources getDefaultResources() {
155        if (DEFAULT_RESOURCES == null) {
156            DEFAULT_RESOURCES = GWT.create(Resources.class);
157        }
158        return DEFAULT_RESOURCES;
159    }
160
161    @Override
162    public void setVisibleOn(final String deviceSizeString) {
163        StyleHelper.setVisibleOn(this, deviceSizeString);
164    }
165
166    @Override
167    public void setHiddenOn(final String deviceSizeString) {
168        StyleHelper.setHiddenOn(this, deviceSizeString);
169    }
170
171    public void setStriped(final boolean striped) {
172        if (striped) {
173            StyleHelper.addEnumStyleName(this, TableType.STRIPED);
174        } else {
175            StyleHelper.removeEnumStyleName(this, TableType.STRIPED);
176        }
177    }
178
179    public void setBordered(final boolean bordered) {
180        if (bordered) {
181            StyleHelper.addEnumStyleName(this, TableType.BORDERED);
182        } else {
183            StyleHelper.removeEnumStyleName(this, TableType.BORDERED);
184        }
185    }
186
187    public void setCondensed(final boolean condensed) {
188        if (condensed) {
189            StyleHelper.addEnumStyleName(this, TableType.CONDENSED);
190        } else {
191            StyleHelper.removeEnumStyleName(this, TableType.CONDENSED);
192        }
193    }
194
195    public void setHover(final boolean hover) {
196        if (hover) {
197            StyleHelper.addEnumStyleName(this, TableType.HOVER);
198        } else {
199            StyleHelper.removeEnumStyleName(this, TableType.HOVER);
200        }
201    }
202}
